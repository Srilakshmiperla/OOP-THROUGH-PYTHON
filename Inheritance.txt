Program for single level Inheritance
class employee1:    #This is parent class
     def __init__(self,name,age,salary):
          self.name = name
          self.age = age
          self.salary = salary
class employee2(employee1):      #This is child class
     def __init__(self,name,age,salary,Id):
          self.name = name
          self.age = age
          self.salary = salary
          self.Id = Id
#Object creation
emp1 = employee1("ABC",18,10000)
print(emp1.name)
print(emp1.age)
print(emp1.salary)


Output:
ABC
18
10000
