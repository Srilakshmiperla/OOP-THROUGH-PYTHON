1. Program for single level Inheritance


class employee1:    #This is parent class
     def __init__(self,name,age,salary):
          self.name = name
          self.age = age
          self.salary = salary
class employee2(employee1):      #This is child class
     def __init__(self,name,age,salary,Id):
          self.name = name
          self.age = age
          self.salary = salary
          self.Id = Id
#Object creation
emp1 = employee1("ABC",18,10000)
print(emp1.name)
print(emp1.age)
print(emp1.salary)


Output:
ABC
18
10000

3. Program for Multiple Inheritance


class person1:
    def __init__(self,name,age,salary):
        self.name = name
        self.age = age
        self.salary = salary
class person2:
    def __init__(self,name,age,salary,Id):
        self.name = name
        self.age = age
        self.salary = salary
        self.Id = Id
class person(person1,person2):
    def __init__(self,name,age,salary,Id):
        self.name = name
        self.age = age
        self.salary = salary
        self.Id = Id

obj1 = person1("ABC",30,100000)

4. Program for Hierarchieal Inheritance

class Person:
    def __init__(self,name,age,salary):
        self.name = name
        self.age = age
        self.salary = salary
class Person1(Person):
    def __init__(self,name,age,salary):
        self.name = name
        self.age = age
        self.salary = salary
class Person2(Person):
        def __init__(self,name,age,salary):
            self.name = name
            self.age = age
            self.salary = salary
obj1 = Person("ABC",25,100000)
obj2 = Person("XYZ",30,1000000)
print(obj1.name)
print(obj1.age)
print(obj2.salary)


Output:
ABC
25
1000000
