from tkinter import *
from tkinter import messagebox
import pyqrcode

ws = Tk()
ws.title("CSE-II  A , B CRT OOPS")
ws.config(bg='yellow')

def generate_QR():
    if len(user_input.get()) != 0:
        global qr, img
        qr = pyqrcode.create(user_input.get())
        img = BitmapImage(data=qr.xbm(scale=8))   # BitmapImage comes from tkinter
        display_code()
    else:
        messagebox.showwarning('warning', 'All Fields are Required')

def display_code():
    img_lbl.config(image=img)
    output.config(text="QR code of " + user_input.get())

lbl = Label(ws, text="Enter message -cse or URL", bg='GREEN')
lbl.pack()

user_input = StringVar()
entry = Entry(ws, textvariable=user_input)
entry.pack(padx=10)

button = Button(ws, text="Gen  CSE A B", width=20, command=generate_QR)
button.pack(pady=20)

img_lbl = Label(ws, bg='green')
img_lbl.pack()

output = Label(ws, text="", bg='pink')
output.pack()

ws.mainloop()
